 12: /**
 13:  *  IP 地理位置查询类 修改自 CoolCode.CN
 14:  *  由于使用UTF8编码 如果使用纯真IP地址库的话 需要对返回结果进行编码转换
 15:  * @author    liu21st <liu21st@gmail.com>
 16:  */
 17: class IpLocation {
 18:     /**
 19:      * QQWry.Dat文件指针
 20:      *
 21:      * @var resource
 22:      */
 23:     private $fp;
 24: 
 25:     /**
 26:      * 第一条IP记录的偏移地址
 27:      *
 28:      * @var int
 29:      */
 30:     private $firstip;
 31: 
 32:     /**
 33:      * 最后一条IP记录的偏移地址
 34:      *
 35:      * @var int
 36:      */
 37:     private $lastip;
 38: 
 39:     /**
 40:      * IP记录的总条数（不包含版本信息记录）
 41:      *
 42:      * @var int
 43:      */
 44:     private $totalip;
 45: 
 46:     /**
 47:      * 构造函数，打开 QQWry.Dat 文件并初始化类中的信息
 48:      *
 49:      * @param string $filename
 50:      * @return IpLocation
 51:      */
 52:     public function __construct($filename = "UTFWry.dat") {
 53:         $this->fp = 0;
 54:         if (($this->fp      = fopen(dirname(__FILE__).'/'.$filename, 'rb')) !== false) {
 55:             $this->firstip  = $this->getlong();
 56:             $this->lastip   = $this->getlong();
 57:             $this->totalip  = ($this->lastip - $this->firstip) / 7;
 58:         }
 59:     }
 60: 
 61:     /**
 62:      * 返回读取的长整型数
 63:      *
 64:      * @access private
 65:      * @return int
 66:      */
 67:     private function getlong() {
 68:         //将读取的little-endian编码的4个字节转化为长整型数
 69:         $result = unpack('Vlong', fread($this->fp, 4));
 70:         return $result['long'];
 71:     }
 72: 
 73:     /**
 74:      * 返回读取的3个字节的长整型数
 75:      *
 76:      * @access private
 77:      * @return int
 78:      */
 79:     private function getlong3() {
 80:         //将读取的little-endian编码的3个字节转化为长整型数
 81:         $result = unpack('Vlong', fread($this->fp, 3).chr(0));
 82:         return $result['long'];
 83:     }
 84: 
 85:     /**
 86:      * 返回压缩后可进行比较的IP地址
 87:      *
 88:      * @access private
 89:      * @param string $ip
 90:      * @return string
 91:      */
 92:     private function packip($ip) {
 93:         // 将IP地址转化为长整型数，如果在PHP5中，IP地址错误，则返回False，
 94:         // 这时intval将Flase转化为整数-1，之后压缩成big-endian编码的字符串
 95:         return pack('N', intval(ip2long($ip)));
 96:     }
 97: 
 98:     /**
 99:      * 返回读取的字符串
100:      *
101:      * @access private
102:      * @param string $data
103:      * @return string
104:      */
105:     private function getstring($data = "") {
106:         $char = fread($this->fp, 1);
107:         while (ord($char) > 0) {        // 字符串按照C格式保存，以\0结束
108:             $data  .= $char;             // 将读取的字符连接到给定字符串之后
109:             $char   = fread($this->fp, 1);
110:         }
111:         return $data;
112:     }
113: 
114:     /**
115:      * 返回地区信息
116:      *
117:      * @access private
118:      * @return string
119:      */
120:     private function getarea() {
121:         $byte = fread($this->fp, 1);    // 标志字节
122:         switch (ord($byte)) {
123:             case 0:                     // 没有区域信息
124:                 $area = "";
125:                 break;
126:             case 1:
127:             case 2:                     // 标志字节为1或2，表示区域信息被重定向
128:                 fseek($this->fp, $this->getlong3());
129:                 $area = $this->getstring();
130:                 break;
131:             default:                    // 否则，表示区域信息没有被重定向
132:                 $area = $this->getstring($byte);
133:                 break;
134:         }
135:         return $area;
136:     }
137: 
138:     /**
139:      * 根据所给 IP 地址或域名返回所在地区信息
140:      *
141:      * @access public
142:      * @param string $ip
143:      * @return array
144:      */
145:     public function getlocation($ip='') {
146:         if (!$this->fp) return null;            // 如果数据文件没有被正确打开，则直接返回空
147:         if(empty($ip)) $ip = get_client_ip();
148:         $location['ip'] = gethostbyname($ip);   // 将输入的域名转化为IP地址
149:         $ip = $this->packip($location['ip']);   // 将输入的IP地址转化为可比较的IP地址
150:                                                 // 不合法的IP地址会被转化为255.255.255.255
151:         // 对分搜索
152:         $l = 0;                         // 搜索的下边界
153:         $u = $this->totalip;            // 搜索的上边界
154:         $findip = $this->lastip;        // 如果没有找到就返回最后一条IP记录（QQWry.Dat的版本信息）
155:         while ($l <= $u) {              // 当上边界小于下边界时，查找失败
156:             $i = floor(($l + $u) / 2);  // 计算近似中间记录
157:             fseek($this->fp, $this->firstip + $i * 7);
158:             $beginip = strrev(fread($this->fp, 4));     // 获取中间记录的开始IP地址
159:             // strrev函数在这里的作用是将little-endian的压缩IP地址转化为big-endian的格式
160:             // 以便用于比较，后面相同。
161:             if ($ip < $beginip) {       // 用户的IP小于中间记录的开始IP地址时
162:                 $u = $i - 1;            // 将搜索的上边界修改为中间记录减一
163:             }
164:             else {
165:                 fseek($this->fp, $this->getlong3());
166:                 $endip = strrev(fread($this->fp, 4));   // 获取中间记录的结束IP地址
167:                 if ($ip > $endip) {     // 用户的IP大于中间记录的结束IP地址时
168:                     $l = $i + 1;        // 将搜索的下边界修改为中间记录加一
169:                 }
170:                 else {                  // 用户的IP在中间记录的IP范围内时
171:                     $findip = $this->firstip + $i * 7;
172:                     break;              // 则表示找到结果，退出循环
173:                 }
174:             }
175:         }
176: 
177:         //获取查找到的IP地理位置信息
178:         fseek($this->fp, $findip);
179:         $location['beginip'] = long2ip($this->getlong());   // 用户IP所在范围的开始地址
180:         $offset = $this->getlong3();
181:         fseek($this->fp, $offset);
182:         $location['endip'] = long2ip($this->getlong());     // 用户IP所在范围的结束地址
183:         $byte = fread($this->fp, 1);    // 标志字节
184:         switch (ord($byte)) {
185:             case 1:                     // 标志字节为1，表示国家和区域信息都被同时重定向
186:                 $countryOffset = $this->getlong3();         // 重定向地址
187:                 fseek($this->fp, $countryOffset);
188:                 $byte = fread($this->fp, 1);    // 标志字节
189:                 switch (ord($byte)) {
190:                     case 2:             // 标志字节为2，表示国家信息又被重定向
191:                         fseek($this->fp, $this->getlong3());
192:                         $location['country']    = $this->getstring();
193:                         fseek($this->fp, $countryOffset + 4);
194:                         $location['area']       = $this->getarea();
195:                         break;
196:                     default:            // 否则，表示国家信息没有被重定向
197:                         $location['country']    = $this->getstring($byte);
198:                         $location['area']       = $this->getarea();
199:                         break;
200:                 }
201:                 break;
202:             case 2:                     // 标志字节为2，表示国家信息被重定向
203:                 fseek($this->fp, $this->getlong3());
204:                 $location['country']    = $this->getstring();
205:                 fseek($this->fp, $offset + 8);
206:                 $location['area']       = $this->getarea();
207:                 break;
208:             default:                    // 否则，表示国家信息没有被重定向
209:                 $location['country']    = $this->getstring($byte);
210:                 $location['area']       = $this->getarea();
211:                 break;
212:         }
213:         if (trim($location['country']) == 'CZ88.NET') {  // CZ88.NET表示没有有效信息
214:             $location['country'] = '未知';
215:         }
216:         if (trim($location['area']) == 'CZ88.NET') {
217:             $location['area'] = '';
218:         }
219:         return $location;
220:     }
