CDataProvider是实现IDataProvider接口的基类。

继承的类主要实现三个方法：fetchData， fetchKeys和calculateTotalItemCount。

public function setPagination($value)
{
    if(is_array($value))
    {
        $pagination=$this->getPagination();
        foreach($value as $k=>$v)
            $pagination->$k=$v;
    }
    else
        $this->_pagination=$value;
}
