190:      *      
191:      *      高中的数学公式咋都忘了涅，写出来
192:      *      正弦型函数解析式：y=Asin(ωx+φ)+b
193:      *      各常数值对函数图像的影响：
194:      *        A：决定峰值（即纵向拉伸压缩的倍数）
195:      *        b：表示波形在Y轴的位置关系或纵向移动距离（上加下减）
196:      *        φ：决定波形与X轴位置关系或横向移动距离（左加右减）

200:     private function _writeCurve() {
201:         $px = $py = 0;
202:         
203:         // 曲线前部分
204:         $A = mt_rand(1, $this->imageH/2);                  // 振幅
205:         $b = mt_rand(-$this->imageH/4, $this->imageH/4);   // Y轴方向偏移量
206:         $f = mt_rand(-$this->imageH/4, $this->imageH/4);   // X轴方向偏移量
207:         $T = mt_rand($this->imageH, $this->imageW*2);  // 周期
208:         $w = (2* M_PI)/$T;
209:                         
210:         $px1 = 0;  // 曲线横坐标起始位置
211:         $px2 = mt_rand($this->imageW/2, $this->imageW * 0.8);  // 曲线横坐标结束位置
212: 
213:         for ($px=$px1; $px<=$px2; $px = $px + 1) {
214:             if ($w!=0) {
215:                 $py = $A * sin($w*$px + $f)+ $b + $this->imageH/2;  // y = Asin(ωx+φ) + b
216:                 $i = (int) ($this->fontSize/5);
217:                 while ($i > 0) {    
218:                     imagesetpixel($this->_image, $px + $i , $py + $i, $this->_color);  // 这里(while)循环画像素点比imagettftext和imagestring用字体大小一次画出（不用这while循环）性能要好很多               
219:                     $i--;
220:                 }
221:             }
222:         }
223:         
224:         // 曲线后部分
225:         $A = mt_rand(1, $this->imageH/2);                  // 振幅        
226:         $f = mt_rand(-$this->imageH/4, $this->imageH/4);   // X轴方向偏移量
227:         $T = mt_rand($this->imageH, $this->imageW*2);  // 周期
228:         $w = (2* M_PI)/$T;      
229:         $b = $py - $A * sin($w*$px + $f) - $this->imageH/2;
230:         $px1 = $px2;
231:         $px2 = $this->imageW;
232: 
233:         for ($px=$px1; $px<=$px2; $px=$px+ 1) {
234:             if ($w!=0) {
235:                 $py = $A * sin($w*$px + $f)+ $b + $this->imageH/2;  // y = Asin(ωx+φ) + b

238:                     imagesetpixel($this->_image, $px + $i, $py + $i, $this->_color);

240:                 }
241:             }
242:         }
243:     }
