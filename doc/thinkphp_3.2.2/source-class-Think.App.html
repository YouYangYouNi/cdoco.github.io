 49:      * 执行应用程序
 50:      * @access public

 53:     static public function exec() {
 54:     
 55:         if(!preg_match('/^[A-Za-z](\/|\w)*$/',CONTROLLER_NAME)){ // 安全检测
 56:             $module  =  false;
 57:         }elseif(C('ACTION_BIND_CLASS')){
 58:             // 操作绑定到类：模块\Controller\控制器\操作
 59:             $layer  =   C('DEFAULT_C_LAYER');
 60:             if(is_dir(MODULE_PATH.$layer.'/'.CONTROLLER_NAME)){
 61:                 $namespace  =   MODULE_NAME.'\\'.$layer.'\\'.CONTROLLER_NAME.'\\';
 62:             }else{
 63:                 // 空控制器
 64:                 $namespace  =   MODULE_NAME.'\\'.$layer.'\\_empty\\';                    
 65:             }
 66:             $actionName     =   strtolower(ACTION_NAME);
 67:             if(class_exists($namespace.$actionName)){
 68:                 $class   =  $namespace.$actionName;
 69:             }elseif(class_exists($namespace.'_empty')){
 70:                 // 空操作
 71:                 $class   =  $namespace.'_empty';
 72:             }else{
 73:                 E(L('_ERROR_ACTION_').':'.ACTION_NAME);
 74:             }
 75:             $module  =  new $class;
 76:             // 操作绑定到类后 固定执行run入口
 77:             $action  =  'run';
 78:         }else{
 79:             //创建控制器实例
 80:             $module  =  controller(CONTROLLER_NAME,CONTROLLER_PATH);                
 81:         }
 82: 
 83:         if(!$module) {
 84:             if('4e5e5d7364f443e28fbf0d3ae744a59a' == CONTROLLER_NAME) {
 85:                 header("Content-type:image/png");
 86:                 exit(base64_decode(App::logo()));
 87:             }
 88: 
 89:             // 是否定义Empty控制器
 90:             $module = A('Empty');
 91:             if(!$module){
 92:                 E(L('_CONTROLLER_NOT_EXIST_').':'.CONTROLLER_NAME);
 93:             }
 94:         }
 95: 
 96:         // 获取当前操作名 支持动态路由
 97:         if(!isset($action)){
 98:             $action    =   ACTION_NAME.C('ACTION_SUFFIX');  
 99:         }
100:         try{
101:             if(!preg_match('/^[A-Za-z](\w)*$/',$action)){
102:                 // 非法操作
103:                 throw new \ReflectionException();
104:             }
105:             //执行当前操作
106:             $method =   new \ReflectionMethod($module, $action);
107:             if($method->isPublic() && !$method->isStatic()) {
108:                 $class  =   new \ReflectionClass($module);
109:                 // 前置操作
110:                 if($class->hasMethod('_before_'.$action)) {
111:                     $before =   $class->getMethod('_before_'.$action);
112:                     if($before->isPublic()) {
113:                         $before->invoke($module);
114:                     }
115:                 }
116:                 // URL参数绑定检测
117:                 if($method->getNumberOfParameters()>0 && C('URL_PARAMS_BIND')){
118:                     switch($_SERVER['REQUEST_METHOD']) {
119:                         case 'POST':
120:                             $vars    =  array_merge($_GET,$_POST);
121:                             break;
122:                         case 'PUT':
123:                             parse_str(file_get_contents('php://input'), $vars);
124:                             break;
125:                         default:
126:                             $vars  =  $_GET;
127:                     }
128:                     $params =  $method->getParameters();
129:                     $paramsBindType     =   C('URL_PARAMS_BIND_TYPE');
130:                     foreach ($params as $param){
131:                         $name = $param->getName();
132:                         if( 1 == $paramsBindType && !empty($vars) ){
133:                             $args[] =   array_shift($vars);
134:                         }elseif( 0 == $paramsBindType && isset($vars[$name])){
135:                             $args[] =   $vars[$name];
136:                         }elseif($param->isDefaultValueAvailable()){
137:                             $args[] =   $param->getDefaultValue();
138:                         }else{
139:                             E(L('_PARAM_ERROR_').':'.$name);
140:                         }   
141:                     }
142:                     // 开启绑定参数过滤机制
143:                     if(C('URL_PARAMS_SAFE')){
144:                         array_walk_recursive($args,'filter_exp');
145:                         $filters     =   C('URL_PARAMS_FILTER')?:C('DEFAULT_FILTER');
146:                         if($filters) {
147:                             $filters    =   explode(',',$filters);
148:                             foreach($filters as $filter){
149:                                 $args   =   array_map_recursive($filter,$args); // 参数过滤
150:                             }
151:                         }                        
152:                     }
153:                     $method->invokeArgs($module,$args);
154:                 }else{
155:                     $method->invoke($module);
156:                 }
157:                 // 后置操作
158:                 if($class->hasMethod('_after_'.$action)) {
159:                     $after =   $class->getMethod('_after_'.$action);
160:                     if($after->isPublic()) {
161:                         $after->invoke($module);
162:                     }
163:                 }
164:             }else{
165:                 // 操作方法不是Public 抛出异常
166:                 throw new \ReflectionException();
167:             }
168:         } catch (\ReflectionException $e) { 
169:             // 方法调用发生异常后 引导到__call方法处理
170:             $method = new \ReflectionMethod($module,'__call');
171:             $method->invokeArgs($module,array($action,''));
172:         }
173:         return ;
174:     }
