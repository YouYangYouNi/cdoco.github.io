158:             $line=isset($trace['line'])?$trace['line']:'unknown line';
159:             $trace_msg='#'.$i.'  '.$fun.' called at ['.$file.':'.$line.']';
160:             if(!empty($trace['args'])){
161:                 ChromePhp::groupCollapsed($trace_msg);
162:                 ChromePhp::log($trace['args']);
163:                 ChromePhp::groupEnd();
164:             }else{
165:                 ChromePhp::log($trace_msg);
166:             }
167:         }
168:         ChromePhp::groupEnd();
169:     }else{
170:         if(method_exists('Behavior\ChromePhp',$type)){
171:             //支持type trace，warn,log,error,group, groupCollapsed, groupEnd等
172:             call_user_func(array('Behavior\ChromePhp',$type),$msg);

175:             call_user_func_array(array('Behavior\ChromePhp','log'),func_get_args());

440:     /**
441:      * converts an object to a better format for logging
442:      *
443:      * @param Object
444:      * @return array
445:      */
446:     protected function _convert($object)
447:     {
448:         // if this isn't an object then just return it
449:         if (!is_object($object)) {
450:             return $object;
451:         }
452: 
453:         //Mark this object as processed so we don't convert it twice and it
454:         //Also avoid recursion when objects refer to each other
455:         $this->_processed[] = $object;
456: 
457:         $object_as_array = array();
458: 
459:         // first add the class name
460:         $object_as_array['___class_name'] = get_class($object);
461: 
462:         // loop through object vars
463:         $object_vars = get_object_vars($object);
464:         foreach ($object_vars as $key => $value) {
465: 
466:             // same instance as parent object
467:             if ($value === $object || in_array($value, $this->_processed, true)) {
468:                 $value = 'recursion - parent object [' . get_class($value) . ']';
469:             }
470:             $object_as_array[$key] = $this->_convert($value);
471:         }
472: 
473:         $reflection = new ReflectionClass($object);
474: 
475:         // loop through the properties and add those
476:         foreach ($reflection->getProperties() as $property) {
477: 
478:             // if one of these properties was already added above then ignore it
479:             if (array_key_exists($property->getName(), $object_vars)) {
480:                 continue;
481:             }

492:             }
493: 
494:             // same instance as parent object
495:             if ($value === $object || in_array($value, $this->_processed, true)) {
496:                 $value = 'recursion - parent object [' . get_class($value) . ']';
497:             }
