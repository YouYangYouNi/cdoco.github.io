214:                 header('Content-Type:application/json; charset=utf-8');

216:             case 'XML'  :
217:                 // 返回xml格式数据
218:                 header('Content-Type:text/xml; charset=utf-8');
219:                 exit(xml_encode($data));
220:             case 'JSONP':
221:                 // 返回JSON数据格式到客户端 包含状态信息
222:                 header('Content-Type:application/json; charset=utf-8');
223:                 $handler  =   isset($_GET[C('VAR_JSONP_HANDLER')]) ? $_GET[C('VAR_JSONP_HANDLER')] : C('DEFAULT_JSONP_HANDLER');
224:                 exit($handler.'('.json_encode($data).');');  
225:             case 'EVAL' :
226:                 // 返回可执行的js脚本
227:                 header('Content-Type:text/html; charset=utf-8');
228:                 exit($data);            
229:             default     :

249:     /**
250:      * 默认跳转操作 支持错误导向和正确跳转
251:      * 调用模板显示 默认为public目录下面的success页面
252:      * 提示页面为可配置 支持模板标签
253:      * @param string $message 提示信息
254:      * @param Boolean $status 状态
255:      * @param string $jumpUrl 页面跳转地址
256:      * @param mixed $ajax 是否为Ajax方式 当数字时指定跳转时间
257:      * @access private
258:      * @return void
259:      */
260:     private function dispatchJump($message,$status=1,$jumpUrl='',$ajax=false) {
261:         if(true === $ajax || IS_AJAX) {// AJAX提交
262:             $data           =   is_array($ajax)?$ajax:array();
263:             $data['info']   =   $message;
264:             $data['status'] =   $status;
265:             $data['url']    =   $jumpUrl;
266:             $this->ajaxReturn($data);
267:         }
268:         if(is_int($ajax)) $this->assign('waitSecond',$ajax);
269:         if(!empty($jumpUrl)) $this->assign('jumpUrl',$jumpUrl);
270:         // 提示标题
271:         $this->assign('msgTitle',$status? L('_OPERATION_SUCCESS_') : L('_OPERATION_FAIL_'));
272:         //如果设置了关闭窗口，则提示完毕后自动关闭窗口
273:         if($this->get('closeWin'))    $this->assign('jumpUrl','javascript:window.close();');
274:         $this->assign('status',$status);   // 状态
275:         //保证输出不受静态缓存影响
276:         C('HTML_CACHE_ON',false);
277:         if($status) { //发送成功信息
278:             $this->assign('message',$message);// 提示信息
279:             // 成功操作后默认停留1秒
280:             if(!isset($this->waitSecond))    $this->assign('waitSecond','1');
281:             // 默认操作成功自动返回操作前页面
282:             if(!isset($this->jumpUrl)) $this->assign("jumpUrl",$_SERVER["HTTP_REFERER"]);
283:             $this->display(C('TMPL_ACTION_SUCCESS'));
284:         }else{
285:             $this->assign('error',$message);// 提示信息
286:             //发生错误时候默认停留3秒
287:             if(!isset($this->waitSecond))    $this->assign('waitSecond','3');
288:             // 默认发生错误的话自动返回上页
289:             if(!isset($this->jumpUrl)) $this->assign('jumpUrl',"javascript:history.back(-1);");
290:             $this->display(C('TMPL_ACTION_ERROR'));
291:             // 中止执行  避免出错后继续执行
292:             exit ;
293:         }
294:     }
