public function lastInsertId ($sequence=NULL)
{
    $value=$this->query('SELECT SCOPE_IDENTITY()')->fetchColumn();
    $value=preg_replace('/[,.]0+$/', '', $value); // issue 2312
    return strtr($value,array(','=>'','.'=>''));
}

获取最后插入id值。 MSSQL不支持序列，所以参数被忽略。
